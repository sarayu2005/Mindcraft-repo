🤖 Scenario 2: The Chat Bot

Goal: Keep the program alive until the user says "bye".

active = True

# "While" Loop: Runs as long as active is True
while active == True:
    msg = input("You: ")
    
    if msg == "bye":
        print("Bot: Closing...")
        active = False # Stops loop
🛒 Scenario 3: The Total Calculator

Goal: Calculate the sum of all items in a cart.

prices = [10, 20, 5]
total = 0

# Loop adds each price to the total
for price in prices:
    total = total + price
    print("Current Total:", total)

print("Final Bill:", total)
